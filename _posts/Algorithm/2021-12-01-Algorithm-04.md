```javascript
const getLengthOfStr = (str) => {
  const arr = [];
  let newStr = "";

  for (let i = 0; i < str.length; i++) {
    newStr = str[i];
    for (let j = i + 1; j < str.length; j++) {
      if (newStr.includes(str[j])) {
        break;
      } else {
        newStr += str[j];
      }
    }
    arr.push(newStr);
  }

  let text = 0;

  for (i = 0; i < arr.length; i++) {
    if (text < arr[i].length) {
      text = arr[i].length;
    }
  }
  return text;
};
```
